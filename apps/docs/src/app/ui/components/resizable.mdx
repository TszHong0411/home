## Usage

```tsx
import { ResizableHandle, ResizablePanel, ResizablePanelGroup } from '@tszhong0411/ui'
```

```tsx
<ResizablePanelGroup direction='horizontal'>
  <ResizablePanel>One</ResizablePanel>
  <ResizableHandle />
  <ResizablePanel>Two</ResizablePanel>
</ResizablePanelGroup>
```
